\Large 3. Mengenoperationen (Paar und Produkt)\\
\\
\normalsize
\textbf{geordnete Paare (a,b)}\\
(a,b) sind Menge der Gestalt $\{\{a\},\{a,b\}\}$\\
\\
$(a,b) \neq (b,a)$\\
$\{a,b\}=\{b,a\}$\\
$(a,b) = (c,d) \Rightarrow a=c und b=d$\\
$\{\{a\},\{a,b\}\}=\{\{c\},\{c,d\}\}$\\
